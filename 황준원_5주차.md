# 백준 1번 5와 6의 차이
- - -
```
#include<iostream>
#include<string>

using namespace std;

int main() {
	string a, b;
	cin >> a >> b;
	string temp = a;
	int min = 0, max = 0;
	for (int i = 0; i < a.size(); ++i) {
		if (a[i] == '5') temp[i] = '6';
		else if (a[i] == '6') a[i] = '5';
	}
	max += stoi(temp), min += stoi(a);
	temp = b;
	for (int i = 0; i < b.size(); ++i) {
		if (b[i] == '5') temp[i] = '6';
		else if (b[i] == '6') b[i] = '5';
	}
	max += stoi(temp), min += stoi(b);
	cout << min << ' ' << max;
}
```

# 백준 2번 좋은 단어
- - -
```
#include<iostream>
#include<string>
#include<stack>
using namespace std;

int main() {
	int n, count = 0;
	cin >> n;
	while (n--) {
		string s;
		cin >> s;
		stack<char> stk;
		for (int i = 0; i < s.size(); ++i) {
			if (!stk.empty() && stk.top() == s[i]) {
				stk.pop();
			}
			else stk.push(s[i]);
		}
		if (stk.empty()) ++count;
	}
	cout << count;
	return 0;
}
```

# 백준 3번 카이사르 암호
- - -
```
#include<iostream>
#include<string>
using namespace std;

int main() {
	string s = "";
	cin >> s;
	for (int i = 0; i < s.size(); ++i) {
		if (int(s[i]) - 3 < int('A')) s[i] += 23;
		else s[i] -= 3; 
	}
	cout << s;
}
```

# 프로그래머스 타겟 넘버
- - -
```
#include <string>
#include <vector>

using namespace std;
int answer = 0;

void dfs(vector<int> numbers,int target,int n, int i){
    if(n == target && i == numbers.size()) ++answer;
    else if (i<numbers.size()){
    dfs(numbers,target,n+numbers[i],i+1);
    dfs(numbers,target,n-numbers[i],i+1);
    }
}
int solution(vector<int> numbers, int target) {
    dfs(numbers,target,0,0);    
    return answer;
}
```
