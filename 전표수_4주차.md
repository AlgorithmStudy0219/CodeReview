# 백준 10703 유성 (11%에서 틀림)
- - -
```
#include <iostream>
#include <algorithm>
using namespace std;

char map[3001][3001] = { 0, };

int main() {
	// #:땅 .:공기 X:유성
	int M, N;
	cin >> M >> N;
	for (int i = 0; i < M; i++) {
		for (int k = 0; k < N; k++) {
			cin >> map[i][k];
		}
	}

	// 각 세로별 .의길이를 알면 최솟값이 만나게되는 시간이다. 그만큼 내리면된다?
	int gap[3001] = { 0, }; //각 세로 라인별 유성과 땅 사이의 거리
	int star[3001] = { 0, }; //각 세로 라인별 유성의 위치
	int min = 999;
	for (int i = 0; i < N; i++) {
		int X_pos = -1;
		int Earth_pos = M - 1;
		for (int k = 0; k < M; k++) {
			if (map[k][i] == 'X') X_pos = k;
			else if (map[k][i] == '#') {
				if (k < Earth_pos) {
					Earth_pos = k;
				}
			}
		}

		star[i] = X_pos;
		if (X_pos != -1) gap[i] = Earth_pos - X_pos - 1;
		else if (X_pos == -1)  gap[i] = 99999;
		//cout << "유성:" << X_pos << " 땅:" << Earth_pos << " gap:" << gap[i] << endl;
		if (min > gap[i]) min = gap[i];
	}


	//cout << min << "초 후 충돌." << endl;
	for (int i = 0; i < N; i++) {
		for (int k = star[i]; k >= 0; k--) {
			if ((k + min) % M != k) {
				map[(k + min) % M][i] = map[k][i];
				map[k][i] = '.';
			}
		}
	}
	//cout << endl << endl;

	//cout << "떨어진 후 맵을 확인합니다. " << endl;
	for (int i = 0; i < M; i++) {
		for (int k = 0; k < N; k++) {
			cout << map[i][k];
		}cout << endl;
	}
}

```

# 프로그래머스 (가장 큰수)
- - -
```
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

bool cmp(string a, string b){
    return a+b > b+a ? true : false;
}

string solution(vector<int> numbers) {
    string answer = "";
    vector<string> arr;
    for(int i=0; i<numbers.size(); i++){
        arr.push_back(to_string(numbers[i]));
    }
    sort(arr.begin(), arr.end(), cmp);
    
    for(int i=0; i<arr.size(); i++){
        answer += arr[i];
    }
    if(arr[0] == "0") answer = "0";
    return answer;
}
```
