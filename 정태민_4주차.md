# 백준 1번 (유성) - 런타임 에러뜸.
- - -
```
#include <iostream>
#include <string.h>

using namespace std;

int main()	{
	int col, row, r, c, high, min = -1;
	bool start = false;
	scanf("%d %d", &col, &row);
	char data[col][row+1];
	for(c=0; c<col; c++)
		cin >> data[c];
		
	for(c=0; c<col; c++)
		for(r=0; r<row; r++)
			if(data[c][r] == 'X')
				high = c;
		
	for(r=0; r<row; r++)	{
		int count = 0;
		for(c=0; c<col; c++)	{
			if(start == false && data[c][r] == 'X')
				start = true;
			if(start == true && data[c][r] == '.')
				count++;
			if(start == true && data[c][r] == '#')	{
				start = false;
				break;
			}
		}
		if(count != 0)	{
			if(min == -1)	min = count;
			else
				if(min > count)	min = count;
		}
	}
	
	for(int c=high; c>=0; c--)	{
		for(r=0; r<row; r++)	{
			if(data[c][r] == 'X')	{
				data[c+min][r] = 'X';           // 이 부분에서 런타임 에러 나는거같은데 해결못함..
				data[c][r] = '.';
			}
		}
	}
	
	for(c=0; c<col; c++)	{
			for(r=0; r<row; r++)
				cout << data[c][r];
			cout << endl;
	}
	return 0; 
}
```
