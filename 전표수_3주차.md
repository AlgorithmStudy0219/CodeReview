vector<int> solution(vector<int> prices) {
    vector<int> answer;
   
    vector<int>::iterator it = prices.begin();
    for(it; it!=prices.end(); it++){
        int count = 0;
        vector<int>::iterator it2 = it + 1;
        for(it2; it2 != prices.end(); it2++){
            if(*(it2) >= *it){
                count++;
            }
            else{
                count++;
                break;
            }
        }
        answer.push_back(count);
    }
    
    return answer;
}

# 4949번 균형잡힌 세상(백준)
- - -
```
#include <iostream>
#include <string>
#include <stack>
using namespace std;


int main() {
	while (1) {
		string input;
		getline(cin, input);
		if (input == ".") break;
		stack<char> stack;

		for (int i = 0; i < input.length(); i++) {
			if (input[i] == '(' || input[i] == ')' || input[i] == '[' || input[i] == ']') {
				if (stack.size() !=0  &&  input[i] == ')' && stack.top() == '(') {
				//	cout << " ( ) 로 닫힘" << endl;
					stack.pop();
				}
				else if (stack.size() != 0 && input[i] == ']' && stack.top() == '[') {
				//	cout << " [ ] 로 닫힘" << endl;
					stack.pop();
				}
				else {
				//	cout << input[i] << "를 stack에 추가" << endl;
					stack.push(input[i]);
				}
			}
		}

		if (stack.size() != 0) {
			cout << "no" << endl;
		}
		else if (stack.size() == 0) {
			cout << "yes" << endl;
		}
	}
}
```


# 1699번 제곱수의 합 (백준)
- - -
```
#include <iostream>
using namespace std;

int dp[100001] = { 0, };

int main() {
	int N;
	cin >> N;
	dp[1] = 1;
	dp[2] = 2;
	dp[3] = 3;
	dp[4] = 1;
	dp[5] = 2;
	for (int i = 6; i <= N; i++) {
		for (int j = 1; j * j <= i; j++) {
			if (dp[i] == 0 || dp[i] > dp[i - (j * j)] + 1) {
				dp[i] = dp[i - (j * j)] + 1;
			}
		}
	}
	cout << dp[N];
}
```



# 1543번 문서검색 (백준)
- - -
```
#include <iostream>
#include <string>
using namespace std;

int main() {
	string str;
	string pattern;

	getline(cin, str);
	getline(cin, pattern);

	int answer = 0;
	bool find = true;
	while (find != false) {
		int index = str.find(pattern);
		if (index == -1) {
			find = false;
		}
		else {
			find = true;
			str = str.substr(index + pattern.length());
			answer++;
		}
	}
	cout << answer;
}
```
