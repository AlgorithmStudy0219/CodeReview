#1 유성

```
#include<iostream>
#include<string>
#include<vector>
#include<algorithm>

using namespace std;
vector<string> arr;

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; ++i) {
		string s;
		cin >> s;
		arr.push_back(s);
	}
	int dis = n, a = 0;
	for (int i = 0; i < m; ++i) {
		int X = -1;
		for (int j = 0; j < n; ++j) {
			if (arr[j][i] == 'X' && arr[j][i] != arr[j+1][i]) X = j;
			if (arr[j][i] == '#' && X != -1) {
				dis = min(dis, j - X);
			}
			a = max(a, X);
		}
	}
	--dis;
	
	for (int i = a; i>=0 ; --i) {
		for (int j = 0; j < m; ++j) {
			if (arr[i][j] == 'X' && arr[i + dis][j] == '.') {
				arr[i + dis][j] = 'X';
				arr[i][j] = '.';
			}
		}
	}
	for (int i = 0; i < n; ++i) {
		cout << arr[i] << '\n';
	}

	return 0;
}
```
#2 유기농 배추
- - -
```
#include<iostream>

using namespace std;
int dx[4] = { 1,-1,0,0 };
int dy[4] = { 0,0,1,-1 };
int arr[50][50] = {};
int visited[50][50] = {};
int t, n, m, k;

void Move(int a,int b) {
	visited[a][b] = 1;
	for (int i = 0; i < 4; ++i) {
		int x = a + dx[i], y = b + dy[i];
		if (x >= 0 && y >= 0 && x < m && y < n) {
			if (arr[x][y] == 1 && visited[x][y] != 1) Move(x, y);
		}
	}
}

int main() {
	cin >> t;
	while (t--) {
		cin >> m >> n >> k;
		for (int i = 0; i < m; ++i) {
			for (int j = 0; j < n; ++j) {
				arr[i][j] = 0;
				visited[i][j] = 0;
			}
		}
		for (int i = 0; i < k; ++i) {
			int a, b;
			cin >> a >> b;
			arr[a][b] = 1;
		}
		int count = 0;
		for (int i = 0; i < m; ++i) {
			for (int j = 0; j < n; ++j) {
				if (visited[i][j] != 1 && arr[i][j] == 1) {
					Move(i, j);
					++count;
				}
			}
		}
		cout << count << '\n';
	}
	return 0;
}
```
