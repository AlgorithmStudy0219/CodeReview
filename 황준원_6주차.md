# 백준 숫자 고르기
- - -
```
#include<iostream>
#include<vector>
#include<algorithm>
#include<set>
using namespace std;

int main() {
	int n, count = 0;
	vector<int> v;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		int a;
		cin >> a;
		v.push_back(a);
	}

	set<int> a;
	pair<set<int>::iterator, bool> check;
	vector<int> b;
	for (int i = 0; i < n; ++i) {
		b.clear();
		b.push_back(v[i]);
		bool s = false;
		for (int j = 0; j < n; ++j) {       // 고른 숫자의 밑의 숫자를 벡터에 넣는걸 반복
			b.push_back(v[b[j] - 1]);
			if (v[b[j] - 1] == i + 1) {     // 벡터에 넣은 숫자가 맨처음 고른 숫자와 같으면 고르기 성공, 종료
				s = true;
				break;
			}	
		}
		if (s) {
			for (int j = 0; j < b.size(); ++j) {    // 고르기 성공한 숫자를 set에 넣음
				check = a.insert(b[j]);
				if (check.second == false) break;   // 중복되면 종료
			}
		}
	}
	cout << a.size() << endl;
	set<int>::iterator it;
	for (it = a.begin(); it != a.end(); it++) {
		cout << *it << endl;
	}
}
```

# 백준 동물원
- - -
```
#include<iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	int* dp = new int[n+1];
	dp[0] = 1, dp[1] = 3;
	for (int i = 2; i <= n; ++i) {
		dp[i] = (dp[i - 1] * 2 + dp[i - 2]) % 9901;
	}
	cout << dp[n];
}
```
