#1 유성
- - -
```
#include<iostream>
#include<string>
#include<vector>
#include<algorithm>

using namespace std;
vector<string> arr;

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; ++i) {
		string s;
		cin >> s;
		arr.push_back(s);
	}
	int dis = n, a = 0;		// dis는 운석과 땅의 최소거리, a는 운석의 튀어나온 부분
	for (int i = 0; i < m; ++i) {
		int X = -1;
		for (int j = 0; j < n; ++j) {
			if (arr[j][i] == 'X' && arr[j][i] != arr[j+1][i]) X = j;
			if (arr[j][i] == '#' && X != -1) {
				dis = min(dis, j - X);
			}
			a = max(a, X);
		}
	}
	--dis;
	
	for (int i = a; i >= 0 ; --i) {		// 운석의 끝부터 시작까지
		for (int j = 0; j < m; ++j) {
			if (arr[i][j] == 'X' && arr[i + dis][j] == '.') {	// 운석부분과 공기부분을 바꿔줌
				arr[i + dis][j] = 'X';
				arr[i][j] = '.';
			}
		}
	}
	for (int i = 0; i < n; ++i) {
		cout << arr[i] << '\n';
	}

	return 0;
}
```
#2 유기농 배추
- - -
```
#include<iostream>

using namespace std;
int dx[4] = { 1,-1,0,0 };
int dy[4] = { 0,0,1,-1 };
int arr[50][50] = {};
int visited[50][50] = {};
int t, n, m, k;

void Move(int a,int b) {	// visited를 1로 만들고 방문 안하고 이어져있는지 확인
	visited[a][b] = 1;
	for (int i = 0; i < 4; ++i) {
		int x = a + dx[i], y = b + dy[i];
		if (x >= 0 && y >= 0 && x < m && y < n) {
			if (arr[x][y] == 1 && visited[x][y] != 1) Move(x, y);
		}
	}
}

int main() {
	cin >> t;
	while (t--) {
		cin >> m >> n >> k;
		for (int i = 0; i < m; ++i) {
			for (int j = 0; j < n; ++j) {
				arr[i][j] = 0;
				visited[i][j] = 0;
			}
		}
		for (int i = 0; i < k; ++i) {
			int a, b;
			cin >> a >> b;
			arr[a][b] = 1;
		}
		int count = 0;
		for (int i = 0; i < m; ++i) {
			for (int j = 0; j < n; ++j) {
				if (visited[i][j] != 1 && arr[i][j] == 1) {
					Move(i, j);
					++count;
				}
			}
		}
		cout << count << '\n';
	}
	return 0;
}

#3 문자열 잘라내기 (시간초과)
- - -
```
#include<iostream>
#include<algorithm>
#include<string>
#include<vector>

using namespace std;

int main() {
	int r, c, count = 0;
	cin >> r >> c;
	vector<string> line;
	for (int i = 0; i < r; ++i) {
		string s;
		cin >> s;
		line.push_back(s);
	}

	vector<string> row;
	vector<string>::iterator it;
	for (int i = 0; i < c; ++i) {
		string s = "";
		for (int j = 1; j < r; ++j) {
			s += line[j][i];
		}
		row.push_back(s);
	}
	for (int j = 0; j < r; ++j) {
		for (int i = 0; i < row.size(); ++i) {
			it = find(row.begin()+i+1, row.end(), row[i]);
			if (it != row.end()) {
				cout << count;
				return 0;
			}
		}
		++count;
		for (int i = 0; i < row.size(); ++i) {
			row[i] = row[i].substr(1);
		}
	}
}
