#include<iostream>
#include<string>

using namespace std;

int main(){
	int n = 0, m = 0;
	string P;
	string S;
	cin >> n >> m >> S;
	int count = 0;
	for (int i = 0; i < m; ++i) {
		if (S[i] == 'I' && S[i + 1] == 'O') {
			int Ocount = 0;
			while (S[i + 1] == 'O' && S[i + 2] == 'I') {
				i += 2;
				Ocount += 1;
				if (Ocount == n && S[i] == 'I') ++count, --Ocount;
			}
		}
	}
	cout << count;
}
